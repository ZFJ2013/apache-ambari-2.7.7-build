    getMastersSubmenu: function getMastersSubmenu(context, action, components) {
      var serviceName = context.get('service.serviceName'),
          groups = context.get('service.masterComponentGroups') || [],
          allItem = {
        action: action,
        context: Object.assign({
          label: Em.I18n.t('services.service.allComponents')
        }, components ? {
          components: components,
          hosts: groups.mapProperty('hosts').reduce(function (acc, groupHosts) {
            return [].concat(_toConsumableArray(acc), _toConsumableArray(groupHosts));
          }, []).uniq()
        } : {
          serviceName: serviceName
        }),
        disabled: false
      },
          groupItems = groups.map(function (group) {
        return {
          action: action,
          context: {
            label: group.title,
            hosts: group.hosts,
            components: components || group.components,
            serviceName: serviceName
          },
          disabled: false,
          tooltip: group.title
        };
      });
      return [allItem].concat(_toConsumableArray(groupItems));
    }
